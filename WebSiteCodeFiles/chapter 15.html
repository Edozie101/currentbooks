on getRocketPosition(currentPosition, currentSpeed, currentMass, 
\
massBurnRate, fuelSpeed, gravity, time)
 --! ARGUMENTS: 
currentPosition, currentSpeed, currentMass, 
 --! massBurnRate 
(mass / time), fuelSpeed , gravity, time (all numbers)
 --! 
RETURNS: position of rocket after time
 set MassBurned to 
massBurnRate * time
 set speed to getRocketSpeed(currentSpeed, 
currentMass, massBurned, fuelSpeed, gravity, time)
 return 
currentPosition + speed * time
end 
 
on getRocketSpeed(currentSpeed, currentMass, massBurned, 
fuelSpeed, gravity, time)
 --! ARGUMENTS: currentSpeed, currentMass, 
massBurned, fuelSpeed , gravity, time (all numbers)
 --! 
RETURNS: speed of rocket after time
 return currentSpeed 
+ fuelSpeed * log (currentMass/(currentMass-massBurned)) - gravity 
* time
end
