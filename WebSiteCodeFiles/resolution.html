on resolveSpinningPoolCushionCollision (velocity, angVel, 
efficiency, radius, norm, prop)
 --! ARGUMENTS: velocity 
(2D vector), angVel (number), efficiency, radius (numbers), 

 --! norm (2D vector), prop (number between 0 and 1)
 
--! RETURNS: a list of velocity, angVelocity for an object colliding 
linearly with a 
 --! fixed wall and losing a proportion 
of its angular velocity as a result
 
 -- assumes n points 
towards the ball center
 -- first calculate change in spin
 
set phi to prop*angVel
 set tang to clockwiseNormal(norm)
 
set u to resolvePoolCushionCollision(velocity-0.4*phi*radius*radius*tang,efficiency,norm)
 
set om to angVel-phi
 
 return [u,om]
end 

 on resolvePoolCushionCollision 
(velocity, efficiency, norm)
 --! ARGUMENTS: velocity (2D 
vector), efficiency (number), 
 --! norm (2D vector)
 
--! RETURNS: the resultant velocity for an object colliding 
linearly with
 --! a fixed wall and not changing its angular 
velocity
 set sq to sqrt(efficiency)
 set un to componentVector(velocity,norm)
 
set ut to velocity-un
 set vn to -un*sq
 return ut+vn
end 

 on resolvePoolCollision 
(vel1, vel2, norm)
 --! ARGUMENTS: vel1, vel2, norm (2D vectors)
 
--! RETURNS: a list of the velocities of two objects colliding 
elastically
 --! with no change of angular momentum
 
 
set u to vel1-vel2
 set un to componentVector(u,norm)
 
set ut to u-un
 return [ut+vel2, un+vel2]
end
