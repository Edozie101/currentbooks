 on moment (v, 
p)
 --! ARGUMENTS: v and p (both 2d vectors) 
 --! RETURNS: 
the moment of v about the origin for a particle at p
 set 
n to clockwiseNormal(v)
 
return dotProd(n, p)
end 

 on clockwise p, v
 --! ARGUMENTS: 
p, v (2D vectors)
 --! RETURNS: +1 or -1
 
 -- if obj 
starts at p and has velocity v, this function 
 -- returns 
1 if it is moving clockwise about the origin, -1 otherwise
 
n=clockwiseNormal(p)
 
if component(v,n)>0 
then return 1
 return -1
end
 
on totalKineticEnergy vlist, mlist
 --! ARGUMENTS: vlist 
(a list [velocity, angular velocity]), mlist (a list 
[mass, moment 
of inertia])
 --! RETURNS: the total kinetic energy 
of the object
 v=mag(vlist[1])
 
return 0.5*(mlist[1]*v*v+mlist[2]*vlist[2]*vlist[2])
end
